% ---- Verification methods legend --------------------------------------------
\subsubsection*{Compliance Status Legend}
\begin{itemize}[nosep]
    \item \compliant{} -- \textbf{Compliant}: requirement is fully met.
    \item \partlycompliant{} -- \textbf{Partially Compliant}: requirement is
          partially met; a mitigation plan is in place.
    \item \noncompliant{} -- \textbf{Non-Compliant}: requirement is not met;
          explain why and what actions are planned.
    \item \notapplicable{} -- \textbf{Not Applicable}: requirement does not
          apply to this design; provide justification.
\end{itemize}
